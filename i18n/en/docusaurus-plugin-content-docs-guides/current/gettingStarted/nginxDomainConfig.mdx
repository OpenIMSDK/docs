# Nginx Domain Configuration

## 1. Prerequisites üõ†Ô∏è

- **open-im-server** and **chat** have been successfully started.
- **Nginx** is installed with SSL modules enabled.
- Two domains and their SSL certificates have been successfully obtained, e.g., `web.xx.xx` (for the IM server and Web client) and `admin.xx.xx` (for the admin panel).
- Ports 443 and 80 are open.

## 2. Domain Configuration Template üìù

> üöÄ **Tip**: Be sure to replace the actual domain names, SSL certificate paths, and SSL keys with your own values.

```nginx
# open-im-server and chat deployment addresses and ports
upstream msg_gateway {
    # IM Message server address. Multiple addresses can be specified based on deployment
    server 127.0.0.1:10001;
}
upstream im_api {
    # IM Group user API server address. Multiple addresses can be specified based on deployment
    server 127.0.0.1:10002;
}
upstream im_chat_api {
    # IM Business version login and registration server address. Multiple addresses can be specified based on deployment
    server 127.0.0.1:10008;
}
upstream im_admin_api {
    # Admin address for the commercial version. Multiple units can be specified based on deployment
    server 127.0.0.1:10009;
}
upstream minio_s3_2 {
    # Minio address. Multiple modules can be assigned based on deployment
    server 127.0.0.1:10005;
}
upstream pc_web {
    # PC web address
    server 127.0.0.1:11001;
}
upstream openim_admin {
    # Admin backend address
    server 127.0.0.1:11002;
}

# Example domain configuration for "web.xx.xx"
server {
    listen 443 ssl; # Listen on port 443
    server_name web.xx.xx;  # Your domain name
    # SSL certificate paths
    ssl_certificate /usr/local/nginx/conf/ssh/web.xx.xx_bundle.pem;
    ssl_certificate_key /usr/local/nginx/conf/ssh/web.xx.xx.key;

    # Enable gzip compression
    gzip on;
    gzip_min_length 1k;
    gzip_buffers 4 16k;
    gzip_comp_level 2;
    gzip_types text/plain application/javascript application/x-javascript text/css application/xml text/javascript application/x-httpd-php image/jpeg image/gif image/png application/wasm;
    gzip_vary off;
    gzip_disable "MSIE [1-6]\.";

    default_type application/wasm;

    # Proxy configuration for the root path
    location / {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://pc_web/;
    }

    # Proxy configuration for msg_gateway
    location /msg_gateway {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://msg_gateway/;
    }

    # Proxy configuration for the API
    location ^~/api/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_set_header X-Request-Api $scheme://$host/api;
        proxy_pass http://im_api/;
    }

    # Proxy configuration for chat
    location ^~/chat/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://im_chat_api/;
    }

    # Proxy configuration for MinIO API
    location ^~/im-minio-api/ {
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 300;
        proxy_http_version 1.1;
        proxy_set_header Connection "";
        chunked_transfer_encoding off;
        proxy_pass http://minio_s3_2/;
    }
}

# Example domain configuration for "admin.xx.xx"
server {
    listen 443 ssl; # Listening on port 443
    server_name admin.xx.xx;  # Your admin domain name
    # SSL certificate paths
    ssl_certificate /usr/local/nginx/conf/ssh/admin.xx.xx_bundle.pem;
    ssl_certificate_key /usr/local/nginx/conf/ssh/admin.xx.xx.key;

    # Enable gzip compression
    gzip on;
    gzip_min_length 1k;
    gzip_buffers 4 16k;
    gzip_comp_level 2;
    gzip_types text/plain application/javascript application/x-javascript text/css application/xml text/javascript application/x-httpd-php image/jpeg image/gif image/png application/wasm;
    gzip_vary off;
    gzip_disable "MSIE [1-6]\.";

    default_type application/wasm;

    # Proxy configuration for the root path
    location / {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://openim_admin/;
    }

    # Proxy configuration for msg_gateway
    location /msg_gateway {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://msg_gateway/;
    }

    # Proxy configuration for the API
    location ^~/api/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_set_header X-Request-Api $scheme://$host/api;
        proxy_pass http://im_api/;
    }

    # Proxy configuration for chat
    location ^~/chat/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://im_chat_api/;
    }

    # Proxy configuration for the admin backend
    location ^~/complete_admin/ {
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header X-real-ip $remote_addr;
        proxy_set_header X-Forwarded-For $remote_addr;
        proxy_pass http://im_admin_api/;
    }
}

# HTTP to HTTPS redirection
server {
    listen 80;
    server_name web.xx.xx;
    rewrite ^(.*)$ https://$host$1 permanent;
}
```

## 3. MinIO Configuration üóÑÔ∏è

- **Source Code Deployment**: Modify the `externalAddress` in the `config/minio.yml` file to `"https://web.xx.xx/im-minio-api"`.
- **Docker Deployment**: Modify the `MINIO_EXTERNAL_ADDRESS` in the `.env` file to `"https://web.xx.xx/im-minio-api"`.

## 4. Start Nginx üöÄ

Run the command `nginx -s reload` to reload the Nginx configuration and apply the new domain settings.

## 5. Access Verification üîç

- Access the IM web client: `https://web.xx.xx`
- Access the admin panel: `https://admin.xx.xx`
